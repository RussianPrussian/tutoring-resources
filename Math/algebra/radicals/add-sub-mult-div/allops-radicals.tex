\begin{enumerate}

\item $\sqrt{xy}  + 2\sqrt{xyz} + 2\sqrt{xyz} - 4\sqrt{xy}$

\item $\sqrt{8} + 2\sqrt{8} - \sqrt{2} + \sqrt{3} + \sqrt{9} + 1 - 10\sqrt{3}$
